/description ( Provinces, Districts, Sub-districts, Villages, and Ministry of Health Districts) def
{  db:
   /server (localhost) def
   /dbname (iridb) def
   :db
   (select gid,province_n,district_n,dsd_n,gnd_n,gnd_c,area,the_geom from sl_gndmrg order by gid) (gid) openqueryby
        province_n
            /long_name (Province Name) def
        CopyStream
        name exch def
        district_n
           /long_name (District Name) def
        CopyStream
        name exch def
        dsd_n
           /long_name (Sub-District Name) def
        CopyStream
        name exch def
        gnd_n
           /long_name (Village Name) def
        CopyStream
        name exch def
        gnd_c
           /long_name (Village Code) def
        CopyStream
        name exch def
        area
           /long_name (Area) def
           /units (m2) def
        CopyStream
        name exch def
        the_geom
             /long_name (Geometry) def
        CopyStream
        name exch def
}defasvarcachesilent
