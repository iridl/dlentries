/Districts
/description ( Provinces and Districts) def
{  db:
   /server (localhost) def
   /dbname (iridb) def
   :db
   (select gid,gid as geom_id,name1,name2,the_geom,(name2||', '|| name1) as label from indonesia_dist order by gid) (gid) openqueryby
        geom_id
            /long_name (Geometry_id) def
        CopyStream
        name exch def
        name1
           /long_name (Province Name) def
        CopyStream
        name exch def
        name2
            /long_name (District Name) def
        CopyStream
        name exch def
        label
        CopyStream
        name exch def
        the_geom
             /long_name (District Geometry) def
        CopyStream
        name exch def
}defasvarcachesilent
